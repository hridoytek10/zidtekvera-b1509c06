const fs   = require(&apos;fs-extra&apos;)
const path = require(&apos;path&apos;)
 
async function injectTemplate(buildDir, data) {
  const files = getAllFiles(buildDir)  // walk all files recursively
  const exts  = [&apos;.js&apos;,&apos;.jsx&apos;,&apos;.ts&apos;,&apos;.tsx&apos;,&apos;.json&apos;,&apos;.html&apos;,&apos;.css&apos;,&apos;.md&apos;]
  for (const file of files) {
    if (!exts.includes(path.extname(file))) continue
    let content = await fs.readFile(file, &apos;utf8&apos;)
    content = content
      .replace(/__USER_NAME__/g,      sanitize(data.name))
      .replace(/__USER_BIO__/g,       sanitize(data.bio))
      .replace(/__USER_ADDRESS__/g,   sanitize(data.address))
      .replace(/__HERO_TITLE__/g,     sanitize(data.heroTitle))
      .replace(/__HERO_SUB__/g,       sanitize(data.heroSub))
      .replace(/__CTA_LABEL__/g,      sanitize(data.ctaLabel))
      .replace(/__CTA_LINK__/g,       data.ctaLink)
      .replace(/__SECTION1_TITLE__/g, sanitize(data.section1Title))
      .replace(/__SECTION1_TEXT__/g,  sanitize(data.section1Text))
      .replace(/__SECTION2_TITLE__/g, sanitize(data.section2Title))
      .replace(/__SECTION2_TEXT__/g,  sanitize(data.section2Text))
      .replace(/__GITHUB_URL__/g,     data.github)
      .replace(/__LINKEDIN_URL__/g,   data.linkedin)
      .replace(/__PROFILE_IMAGE__/g,  &apos;/profile.jpg&apos;)
    await fs.writeFile(file, content, &apos;utf8&apos;)
  }
}
5.3  lib/createGithubRepo.js
async function createGithubRepo(repoName) {
  const res = await fetch(&apos;https://api.github.com/user/repos&apos;, {
    method: &apos;POST&apos;,
    headers: {
      Authorization: `token ${process.env.GITHUB_TOKEN}`,
      &apos;Content-Type&apos;: &apos;application/json&apos;,
    },
    body: JSON.stringify({ name: repoName, private: false, auto_init: false })
  })
  const data = await res.json()
  return data.clone_url   // https://github.com/user/repo-name.git
}

async function pushToGithub(buildDir, repoUrl) {
  const git = simpleGit(buildDir)
  await git.init()
  await git.addConfig(&apos;user.name&apos;,  process.env.GIT_USER_NAME)
  await git.addConfig(&apos;user.email&apos;, process.env.GIT_USER_EMAIL)
  await git.add(&apos;.&apos;)
  await git.commit(&apos;Initial commit&apos;)
  await git.addRemote(&apos;origin&apos;, repoUrl)
  await git.push(&apos;origin&apos;, &apos;main&apos;, [&apos;--set-upstream&apos;])
}

7. Vercel Auto Deploy Setup
7.1  Method A — Manual Import (Simplest)
Go to vercel.com → Add New → Project
Import the newly created GitHub repo
Vercel auto-detects Next.js and sets build config
Click Deploy — site goes live at https://repo-name.vercel.app
All future pushes to main auto-deploy
7.2  Method B — Vercel API (Fully Automated, Recommended)
Call this after pushToGithub() to link and deploy the repo automatically:
async function linkVercel(repoName, githubUser) {
  await fetch(&apos;https://api.vercel.com/v1/projects&apos;, {
    method: &apos;POST&apos;,
    headers: {
      Authorization: `Bearer ${process.env.VERCEL_TOKEN}`,
      &apos;Content-Type&apos;: &apos;application/json&apos;,
    },
    body: JSON.stringify({
      name: repoName,
      gitRepository: {
        type: &apos;github&apos;,
        repo: `${githubUser}/${repoName}`,
      },
      framework: &apos;nextjs&apos;,
    })
  })
  return `https://${repoName}.vercel.app`
}

9. Step-by-Step Build Order in Cursor
Phase 1 — Setup
Run: npx create-next-app@latest landing-page-builder
Select: App Router = Yes, Tailwind = Yes
Run: npm install fs-extra formidable simple-git uuid
Create all folders: /components, /lib, /template
Create .env.local and add to .gitignore
Phase 2 — Template
Place your Next.js landing page inside /template
Add all __PLACEHOLDER__ tags inside the template JSX
Verify template builds: cd template &amp;&amp; npm install &amp;&amp; npm run build
Confirm /template/public/ folder exists (profile image goes here)
Phase 3 — Backend
Create lib/injectTemplate.js — test with dummy data first
Create lib/createGithubRepo.js — test with real GitHub token
Create lib/pushToGithub.js — test with a real repo
Create lib/cleanupTemp.js
Create app/api/generate/route.js — wire all lib functions together
Test full API with Postman: POST /api/generate with FormData
Phase 4 — Frontend
Build StepWelcome.jsx — hero text, feature cards, start button
Build StepPersonal.jsx — photo upload + name, bio, address, socials
Build StepContent.jsx — hero, CTA, section editors
Build StepPreview.jsx — live preview with desktop/mobile toggle
Build StepDeploying.jsx — pipeline bar + terminal log
Build StepDone.jsx — URL display, open/copy, share uses, redeploy
Wire all screens in app/page.jsx with useState
Phase 5 — Connect &amp; Test
Connect form submit to POST /api/generate
Test full end-to-end: form → preview → deploy → live URL
Test edge cases: wrong image type, large file, missing fields
Verify temp folders are deleted after each build
Phase 6 — Deploy the Builder App
Push your builder project to its own GitHub repo
Import that repo into Vercel
Add all .env.local variables in Vercel → Settings → Environment Variables
Deploy and test the live builder
